using System;

class Program
{
    static void Main()
    {
        Console.WriteLine("1 для дюймы > см или 2 для см > дюймы:");
        int choice = Convert.ToInt32(Console.ReadLine());

        Console.WriteLine("Введите расстояние:");
        double distance = Convert.ToDouble(Console.ReadLine());

        if (choice == 1)
        {
            double dCm = distance * 2.54;
            Console.WriteLine($"Расстояние в сантиметрах: {dCm} см");
        }
        else if (choice == 2)
        {
            double dInch = distance / 2.54;
            Console.WriteLine($"Расстояние в дюймах: {dInch} дюймов");
        }
        else
        {
            Console.WriteLine("Некорректный выбор. Пожалуйста, введите 1 или 2.");
        }
    }
}

using System;

class Program
{
    static void Main()
    {
        Console.WriteLine("Выберите вариант расчета (1, 2 или 3):");
        int v = Convert.ToInt32(Console.ReadLine());
        int d, ks, ko;
        int otv = 0;

        if (v == 1)
        {
            Console.WriteLine("Введите желаемый доход:");
            d = Convert.ToInt32(Console.ReadLine());
            Console.WriteLine("Введите количество опозданий:");
            ko = Convert.ToInt32(Console.ReadLine());

            otv = (d / 50) * 100;
            if (d % 50 != 0)
            {
                otv += 100;
            }
            if (ko % 3 == 0)
            {
                otv += ko / 3 * 20;
            }
        }
        else if (v == 2)
        {
            Console.WriteLine("Введите количество строк кода:");
            ks = Convert.ToInt32(Console.ReadLine());
            Console.WriteLine("Введите желаемый доход:");
            d = Convert.ToInt32(Console.ReadLine());

            if (ks % 100 != 0)
            {
                otv = (ks / 100 + 1) * 50;
            }
            else
            {
                otv = ks / 100 * 50;
            }

            int dko = 0;
            bool unreachable = true;
            while (otv < d)
            {
                dko += 20;
                otv += 100;
                if (dko >= 40100)
                {
                    Console.WriteLine("ERROR");
                    unreachable = false;
                    break;
                }
            }
            if (unreachable)
            {
                otv = dko / 20;
            }
        }
        else if (v == 3)
        {
            Console.WriteLine("Введите количество строк кода:");
            ks = Convert.ToInt32(Console.ReadLine());
            Console.WriteLine("Введите количество опозданий:");
            ko = Convert.ToInt32(Console.ReadLine());

            otv = (ks / 100) * 50;
            otv -= (ko / 3) * 20;
            otv = Math.Max(otv, 0);
        }
        else
        {
            Console.WriteLine("ERROR");
            return;
        }

        Console.WriteLine("Результат: " + otv);
    }
}
